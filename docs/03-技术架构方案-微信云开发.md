### 2.2 统一中间层设计

每个云函数入口采用统一结构：

```javascript
// cloudfunctions/job/index.js — 统一中间层模式
const cloud = require('wx-server-sdk')
cloud.init({ env: cloud.DYNAMIC_CURRENT_ENV })

// 公共模块
const { checkAuth } = require('./lib/auth')        // 身份+角色校验
const { validateParams } = require('./lib/validate') // 参数校验
const { logOperation } = require('./lib/logger')    // 操作日志

const db = cloud.database()

exports.main = async (event, context) => {
  const { action, data } = event
  const wxContext = cloud.getWXContext()

  // 1. 统一返回格式
  const res = (code, message, data = null) => ({ code, message, data })

  try {
    // 2. 身份校验（获取当前用户角色）
    const auth = await checkAuth(wxContext)
    if (!auth) return res(401, '未登录')

    // 3. 参数校验
    const schema = getSchema(action)
    if (schema) {
      const valid = validateParams(schema, data)
      if (!valid) return res(400, '参数错误')
    }

    // 4. 执行具体业务逻辑
    let result
    switch (action) {
      case 'list':
        result = await listJobs(data, auth)
        break
      case 'search':
        result = await searchJobs(data)
        break
      case 'getDetail':
        result = await getJobDetail(data)
        break
      case 'publish':
        result = await publishJob(data, auth)
        break
      case 'update':
        result = await updateJob(data, auth)
        break
      case 'close':
        result = await closeJob(data, auth)
        break
      default:
        return res(404, '未知操作')
    }

    // 5. 操作日志（写操作用）
    if (['publish', 'update', 'close'].includes(action)) {
      await logOperation({
        action, targetId: result._id,
        operator: auth.accountId, detail: data
      })
    }

    return res(0, 'success', result)
  } catch (err) {
    console.error('[job]', err)
    return res(500, '服务器错误')
  }
}
```

> 3. 识别失败时，展示文字引导按钮 + 客服电话

```javascript
const INTENT_MAP = {
  '找工作': { type: 'navigate', url: '/pages/jobs/jobs' },
  '找活':   { type: 'navigate', url: '/pages/jobs/jobs' },

  '报名':   { type: 'navigate', url: '/pages/apply-record/apply-record' },
  '签到':   { type: 'navigate', url: '/pages/signin/signin' },
  '工资':   { type: 'navigate', url: '/pages/wage/wage' },
  '个人中心': { type: 'navigate', url: '/pages/profile/profile' },
  '我的':   { type: 'navigate', url: '/pages/profile/profile' },
  '返回':   { type: 'back' },
  '上一页':  { type: 'back' },
}

function matchIntent(text) {
  for (const [keyword, action] of Object.entries(INTENT_MAP)) {
    if (text.includes(keyword)) {
      return action
    }
  }
  return null // 未匹配 → 展示引导按钮
}
```

## 五、签到二维码方案

### 5.1 技术方案：wxacode.getUnlimited

| 维度 | 方案详情 |
|------|---------|
| **生成方式** | 云函数调用 `wxacode.getUnlimited` 生成微信小程序码 |
| **携带参数** | scene 参数携带 jobseekerId + 时间戳，加密不可篡改 |
| **使用方式** | 劳动者展示码 → 企业微信扫码 → 云函数校验 → 写入签到记录 |
| **费用** | **免费**（云调用配额内） |
| **安全机制** | ① 只能用微信扫码 ② scene参数加密不可篡改 ③ 扫码时获取企业openid做身份校验 ④ 校验是否已被该企业录取 ⑤ 校验当日是否已签到 |

### 5.2 签到流程

```
劳动者展示小程序码
    ↓
企业管理员微信扫码
    ↓
云函数 checkin.scan:
  1. 解析 scene → jobseekerId + timestamp
  2. 获取扫码者 openid
  3. 校验扫码者是否为该企业管理员
  4. 校验该求职者是否已被该企业录取
  5. 校验该求职者今日是否已签到（防重复）
  6. 校验岗位是否在招聘期内
    ↓
校验通过 → 写入 checkins 记录 → 返回签到成功
校验失败 → 返回错误提示（"未录取"/"已签到"/"非本企业人员"）
```

### 5.3 前端展示

```javascript
// 生成签到二维码（云函数）
```

## 七、安全与权限

### 7.1 数据安全策略

| 数据类型 | 处理方式 |
|---------|---------|
| **手机号** | 存储加密（AES-256）+ 脱敏展示（138****5678）+ 哈希查重 |
| **身份证** | 仅存后4位（当前MVP不需要） |
| **电话号码展示** | C端报名阶段：企业只能看到脱敏号；录取后：授权企业看完整号 |
| **云存储文件** | 营业执照仅 Admin 和企业自身可读写 |
| **操作日志** | 所有审核、禁用等敏感操作留痕 |

### 7.2 权限校验规则（云函数内）

```javascript
// shared/auth.js — 角色校验核心
async function checkRole(wxContext, allowedRoles) {
  const { OPENID } = wxContext

  // 查找账号
  const account = await db.collection('accounts').where({
    openid: OPENID, status: 'active'
  }).get()

  if (!account.data.length) return null

  // 检查角色是否在允许列表中
  if (!allowedRoles.includes(account.data[0].role)) {
    throw new PermissionError('无权限')
  }

  // G端：还需校验区域管辖范围
  if (account.data[0].role === 'gov_admin') {
    // 从 admins 表获取管辖 areaIds
  }

  return account.data[0]
}
```

## 十、微信云开发注意事项

### 10.1 开通前准备

1. 打开微信开发者工具 → 新建项目 → 选择「微信云开发」

   - checkins, wage_statements, wage_disputes
   - policies, favorites, categories
   - operation_logs, daily_stats
4. app.json 注册微信同声传译插件
5. 云函数上传并部署

### 10.2 费用预估

| 项目 | 费用 |
|------|------|
| 云开发基础版 | 免费（查当前套餐） |
| wxacode.getUnlimited | 免费 |
| 微信同声传译插件 | 免费 |
| 腾讯云 SMS | ~0.05元/条，预估500元 |

### 10.3 注意事项

| 注意点 | 说明 |
|--------|------|
| **事务** | 报名写入 + 岗位applyCount更新 放在同一事务中 |
| **幂等** | 扫码签到接口需做幂等（同一二维码不可重复签到） |
| **定时任务** | 用云函数定时触发器，每日凌晨写入 `daily_stats` |
| **云函数冷启动** | 配置云函数保留实例（保持常驻）减少首次调用延迟 |
| **短信限额** | 设置日发送上限，防止异常消耗 |
